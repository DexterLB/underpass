\newcommand\gramrow[3]{
    \ifstrempty{#2}{
        % category is empty, we'll leave just the term
        & \phantom{\catdefsep} & \phantom{\termdefsep} & \quad #3
    }{
        % category is not empty
        \ifstrempty{#1}{
            % token is empty, this is an additional category to the previous token
            \phantom{ } & \cataddsep #2
        }{
            % token is not empty
            #1 &\catdefsep #2
        }
        \ifstrempty{#3}{
            % term is empty
            & \phantom{} & \phantom{}
        }{
            % term is not empty
            & \termdefsep & #3
        }
    } \\
}
